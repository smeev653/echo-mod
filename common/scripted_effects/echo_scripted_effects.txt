echo_development_counter = {
	[[on_startup]
		if = {
			limit = {
				ai = no		
			}
			set_variable = {
				which = echo_adm_dev_counter
				value = 0
			}
			set_variable = {
				which = echo_dip_dev_counter
				value = 0
			}
			set_variable = {
				which = echo_mil_dev_counter
				value = 0
			}
		}
	]
	
	#All updates to variables in this section are in province scope.
	[[adm]
		owner = {
			change_variable = {
				which = echo_adm_dev_counter
				value = 1
			}
		}
	]
	[[dip]
		owner = {
			change_variable = {
				which = echo_dip_dev_counter
				value = 1
			}
		}
	]
	[[mil]
		owner = {
			change_variable = {
				which = echo_mil_dev_counter
				value = 1
			}
		}
	]
	owner = {
		set_variable = {
			which = echo_total_dev_counter
			value = 0
		}
		change_variable = {
			which = echo_total_dev_counter
			which = echo_adm_dev_counter
		}
		change_variable = {
			which = echo_total_dev_counter
			which = echo_dip_dev_counter
		}
		change_variable = {
			which = echo_total_dev_counter
			which = echo_mil_dev_counter
		}
	}
}
